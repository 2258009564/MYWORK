# [洛谷P1135]([P1135 奇怪的电梯 - 洛谷](https://www.luogu.com.cn/problem/P1135))
单源最短路，建图后跑 dijk 即可。

```cpp
void solve()
{
    int n, a, b;
    cin >> n >> a >> b;

    vector<int> v(n + 1), vis = v;
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i];
    }

    vector adj(n + 1, vector<int>());
    for (int i = 1; i <= n; i++)
    {
        auto cur = v[i];
        if (i + cur <= n)
        {
            adj[i].push_back(i + cur);
        }
        if (i - cur >= 1)
        {
            adj[i].push_back(i - cur);
        }
    }

    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq;
    pq.push({0, a});

    vis[a] = 1;
    while (pq.size())
    {
        auto [d, u] = pq.top();
        pq.pop();

        if (u == b)
        {
            return void(cout << d);
        }

        for (auto v : adj[u])
        {
            if (!vis[v])
            {
                vis[v] = 1;
                pq.push({d + 1, v});
            }
        }
    }
    cout << -1;
}
```

注意到边权均为1，其实是退化后的bfs。

void solve()
{
    int n, a, b;
    cin >> n >> a >> b;

    vector<int> v(n + 1), vis = v;
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i];
    }

    queue<pair<int, int>> q;
    q.push({a, 0});
    vis[a] = 1;
    
    while (q.size())
    {
        auto [cur, ans] = q.front();
        q.pop();

        if (cur == b)
        {
            return void(cout << ans);
        }

        if (cur + v[cur] <= n and !vis[cur + v[cur]])
        {
            q.push({cur + v[cur], ans + 1});
            vis[cur + v[cur]] = 1;
        }
        if (cur - v[cur] >= 1 and !vis[cur - v[cur]])
        {
            q.push({cur - v[cur], ans + 1});
            vis[cur - v[cur]] = 1;
        }
    }

    cout << -1;
}

void solve()
{
    int n, m;
    cin >> n >> m;
    vector<string> v(n + 1);
    for (int i = 1; i <= n; i++)
    {
        cin >> v[i];
        v[i] = ' ' + v[i];
    }

    vector ans(n + 1, vector<int>(n + 1, 0));

    vector<pair<int, int>> path;

    auto dfs = [&](auto &&self, int x, int y) -> void
    {
        ans[x][y] = -1; // 暂时标记为 -1，表示“正在搜索中”或“已找过”
        path.push_back({x, y});

        for (auto [dx, dy] : vector<pair<int, int>>{{-1, 0}, {1, 0}, {0, -1}, {0, 1}})
        {
            int nx = x + dx, ny = y + dy;

            if (nx >= 1 and nx <= n and ny >= 1 and ny <= n and
                !ans[nx][ny] and v[nx][ny] != v[x][y])
            {
                self(self, nx, ny);
            }
        }
    };

    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            if (!ans[i][j]) // 如果这个格子还没处理过
            {
                path.clear();   // 清空上次的路径
                dfs(dfs, i, j); // 找出当前连通块所有点

                // 3. 统一给这个块里的所有点赋值
                for (auto [px, py] : path)
                {
                    ans[px][py] = path.size(); // 连通块的大小
                }
            }
        }
    }

    while (m--)
    {
        int x, y;
        cin >> x >> y;
        cout << ans[x][y] << endl;
    }
}
