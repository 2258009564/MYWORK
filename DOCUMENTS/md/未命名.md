### H
学习swap函数，也可以直接通过max min函数找最值。
```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a, b;
    cin >> a >> b;
    
    // 做法一
    cout << max(a, b) << endl
         << min(a, b);

    // 做法二
    if (a < b)
    {
        swap(a, b);
    }
    cout << a << endl
         << b;
}
```

### I
可以继续用max函数，也可以放入数组中进行排序。
```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    // 做法一
    int a, b, c;
    cin >> a >> b >> c;
    cout << "The maximum number is : " << max({a, b, c}) << endl
         << "The minimum number is : " << min({a, b, c});

    // 做法二
    vector<int> v(3);
    for (int i = 0; i < 3; i++)
    {
        cin >> v[i];
    }
    sort(v.begin(), v.end());
    cout << "The maximum number is : " << v[2] << endl
         << "The minimum number is : " << v[0];
}
```

### L
想办法从数字中拆出来一个2即可，拆出来之后判断是否是大于0的偶数。

那个输出自行复制吧 太长了
```cpp
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    
    n -= 2;
    if (n > 0 and n % 2 == 0)
    {
        cout << "YES";
    }
    else
    {
        cout << "NO";
    }
}
```
